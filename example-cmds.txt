ansible -m ping localhost                     ##  tests ping module at localhost
# create file
ansible -m file -a "dest=/home/cloud_user/hello state=touch" localhost  ##  file module create hello on localhost
# create directory
ansible -m file -a "dest=/home/cloud_user/NewDirectory state=directory" localhost  ##  file module creates NewDirectory on localhost

##  enforces a desired state , so returns CHANGE if create and SUCCESS if it confirm the file/directory exists
ansible -m file -a "dest=NewDirectory state=absent" localhost  ##  file module delete/confirms it's absent  on localhost
#  fully list of option https://docs.ansible.com/projects/ansible/latest/collections/ansible/builtin/file_module.html
## copy  file 
ansible -m copy -a "src=/home/cloud_user/hello dest=/home/cloud_user/touch2" localhost 
#  edit file  ..  ensure says dave is great?
ansible -m copy -a "dest=/home/cloud_user/hello content='dave is great?'" localhost 
#   check  mode before running     check & diff
ansible -m copy -a "dest=/home/cloud_user/hello content='dave is great'" localhost   --check --diff 
# timezone 
ansible -m timezone -a "name=Asia/Tokyo" localhost
##   inventory defined  with -i   
ansible -i 'sys1,' -m commmand -a 'date'  all  #  run date on all systems listed in -i  with the comma , it thinks its a file
