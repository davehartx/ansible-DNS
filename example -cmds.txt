# create file
ansible -m file -a "dest=hello state=touch" localhost  ##  file module create hello on localhost
# create firectory
ansible -m file -a "dest=NewDirectory state=directory" localhost  ##  file module creates NewDirectory on localhost

##  enforces a desired state , so returns CHANGE if create and SUCCESS if it confirm the file/directory exists
ansible -m file -a "dest=NewDirectory state=absent" localhost  ##  file module delete/confirms it's absent  on localhost
#  fully list of option https://docs.ansible.com/projects/ansible/latest/collections/ansible/builtin/file_module.html
