ansible -b  -m ping localhost                     ##  tests ping module at localhost
# create file
ansible -b -m file -a "dest=/home/cloud_user/hello state=touch" localhost  ##  file module create hello on localhost
# create directory
ansible -b  -m file -a "dest=/home/cloud_user/NewDirectory state=directory" localhost  ##  file module creates NewDirectory on localhost

##  enforces a desired state , so returns CHANGE if create and SUCCESS if it confirm the file/directory exists
ansible -b -m file -a "dest=NewDirectory state=absent" localhost  ##  file module delete/confirms it's absent  on localhost
#  fully list of option https://docs.ansible.com/projects/ansible/latest/collections/ansible/builtin/file_module.html
## copy  file 
ansible -b -m copy -a "src=/home/cloud_user/hello dest=/home/cloud_user/touch2" localhost 
#  edit file  ..  ensure says dave is great?
ansible -b -m copy -a "dest=/home/cloud_user/hello content='dave is great?'" localhost 
#   check  mode before running     check & diff
ansible -b -m copy -a "dest=/home/cloud_user/hello content='dave is great'" localhost   --check --diff
